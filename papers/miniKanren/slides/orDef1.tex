let or$^o$ x y r =
  ocanren {
    fresh a b in
      (nand$^o$ x x a & nand$^o$ y y b & nand$^o$ a b r)
  }

let rec nand$^o$ x y r =
  ocanren {
    (x === true & y === true & r === false) |
    (x === true & y === false & r === true) |
    (x === false & y === true & r === true) |
    (x === false & y === false & r === false) }
\end{lstlisting}
